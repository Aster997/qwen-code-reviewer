CREATE OR REPLACE PROCEDURE process_memory_mapped_file(
    file_path VARCHAR(255),
    output_table VARCHAR(255)
)
AS $$
DECLARE
    file_handle INTEGER;
    file_size BIGINT;
    buffer BYTEA;
    offset BIGINT := 0;
    chunk_size INTEGER := 4096;

    SELECT pg_stat_file(file_path) INTO file_size;

    END LOOP;
